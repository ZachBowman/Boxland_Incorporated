-- last dir right
-- dir right

-- skid counter max
shift down
-- boost on

-- action run
right up
-- action stand
-- skid counter 0
left down

if (left key down)
  {
  if (dir = right and action = run) SKID
  if (dir = right and action = stand and skid_counter < max) SKID

  last_dir = dir
  dir = left

  if (boost on) action = run
  else action = walk
  }

if (right key down)
  {
  last_dir = dir
  dir = right

  if (boost on) action = run
  else action = walk
  }

if (right key up && left key up)
  {  
  if (action = run or walk) action = stand
  skid_counter = 0
  }

if (shift down) boost = on
else boost = off

if (skid_counter < max) skid_counter += 1
